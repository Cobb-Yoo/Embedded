void loop()
{

	if(Serial.available())
	{
		ch = Serial.read();			// read a single letter
		if(ch >= 'a' && ch <= 'z')
		{
			flashSequence(letters[ch-'a']);	
		}
		else if(ch >= 'A' && ch <= 'Z')
		{	
			flashSequence(letters[ch-'A']);
		}
		else if(ch >= '0' && ch <= '9')
		{
			flashSequence(letters[ch-'0']);
		}
		else if(ch == ' ')
		{
			delay(dotDelay*4);		// gap between words
		}
	}
}
